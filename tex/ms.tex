\begin{abstract}
The observed-frame optical spectrum of each extragalactic DESI target
contains a wealth of information beyond its redshift. \fastspecfit{}
(\url{https://fastspecfit.readthedocs.io}) is open-source Python software
that extracts this information by jointly modeling each DESI spectrum and
its broadband photometry using physically motivated stellar continuum and
emission-line templates. The code operates in two modes: \fastphot{} fits
broadband spectral energy distributions alone, while \fastspec{} performs
a simultaneous spectrophotometric fit. Measured quantities include
light-weighted stellar ages, velocity dispersions, stellar masses,
star-formation rates, K-corrections, rest-frame absolute magnitudes and
colors, and integrated fluxes and equivalent widths for more than 40
emission lines spanning the rest-frame UV through near-infrared. We
describe the algorithms, validate the measurements using repeat
observations and comparisons to complementary codes, and present the
\fastspecfit{} value-added catalog for the DESI Data Release 2,
comprising \todo{N} million extragalactic targets.
\end{abstract}

\section{Stellar Continuum Model}
\label{sec:continuum}
%% ============================================================

% Overview of the two fitting modes (fastphot, fastspec) and the
% pipeline sequence.

\subsection{SPS Template Library}
\label{ssec:templates}

\begin{itemize}
  \item FSPS with C3K stellar library (2750--9100~\AA, $R = 3000$), MIST
    isochrones, Chabrier IMF
  \item Non-parametric SFH: 5 lookback-time bins (30~Myr to 13.7~Gyr)
  \item Solar metallicity default; nebular emission included via FSPS grid
  \item Auxiliary templates: Draine \& Li (2007) dust emission, Nenkova et
    al.\ (2008) AGN torus, Vestergaard \& Wilkes (2001) Fe emission
  \item Wavelength grid: 25~km~s$^{-1}$ pixels in 2750--9100~\AA;
    native FSPS grid outside this range
\end{itemize}

\subsection{Dust Attenuation and Emission}
\label{ssec:dust}

\begin{itemize}
  \item Charlot \& Fall (2000) power-law screen:
    $k(\lambda) \propto (\lambda / 5500\,\text{\AA})^{-0.7}$
  \item Single free parameter $\tauv \in [0, 2]$
  \item Energy-balance IR emission: absorbed luminosity reprocessed through
    Draine \& Li (2007) template
\end{itemize}

\subsection{Redshift, IGM, and Flux Scaling}
\label{ssec:zfactors}

\begin{itemize}
  \item Inoue et al.\ (2014) IGM transmission (Ly$\alpha$ forest,
    Lyman-limit systems)
  \item Luminosity distance, flux normalization, mass normalization
    ($\mathcal{M}_0 = 10^{10}\,\mathrm{M_\odot}$)
  \item Pre-computed \texttt{zfactors} array cached at object
    initialization
\end{itemize}

\subsection{Velocity Dispersion}
\label{ssec:vdisp}

\begin{itemize}
  \item Gaussian convolution on $\log\lambda$ grid
    ($\sigma_\mathrm{pix} = \vdisp / 25\,\mathrm{km\,s^{-1}}$)
  \item FFT-based convolution; pre-computed transform cache up to
    500~km~s$^{-1}$
  \item Fitting window: 3800--6000~\AA\ rest-frame
    (Ca\,H\&K, $G$-band, Mg\,\textsc{b}, Fe)
  \item Two-stage fit: $\chi^2$ grid scan then trust-region refinement
  \item Fallback: nominal 250~km~s$^{-1}$ when wavelength coverage is
    insufficient
\end{itemize}

\subsection{Fitting Algorithm}
\label{ssec:fitting}

\begin{itemize}
  \item Variable projection (VARPRO): outer Brent scalar search over
    $\tauv$, inner non-negative least squares (NNLS) over template
    coefficients
  \item Aperture correction: fiber-to-total flux ratio applied before full
    fit
  \item Full spectrophotometric fit: spectra + photometry stacked
    residuals, $\vdisp$ fixed, nebular emission included
  \item Trust-region reflective (TRF) optimizer when $\vdisp$ is a free
    parameter
\end{itemize}

\subsection{Smooth Continuum Correction}
\label{ssec:smooth}

\begin{itemize}
  \item Nonparametric correction for residual flux-calibration errors and
    template mismatch, computed after SED subtraction
  \item Sliding-window robust median + quadratic spline fit per camera
  \item Added to SED model before emission-line fitting
\end{itemize}

\subsection{Uncertainties}
\label{ssec:continuum_uncertainties}

\begin{itemize}
  \item Monte Carlo: $N = 50$ independent noise realizations of spectra
    and photometry
  \item Uncertainties reported for $\tauv$, $\vdisp$ (when fitted),
    $D_n(4000)$, luminosities, and continuum fluxes
\end{itemize}


%% ============================================================
\section{Emission-Line Model}
\label{sec:emlines}
%% ============================================================

\subsection{Line List and Kinematic Groups}
\label{ssec:linelist}

\begin{itemize}
  \item $>40$ lines from rest-frame UV through near-IR
  \item Lines tied in kinematic groups (shared $v_\mathrm{shift}$,
    $\sigma$) with flexible amplitude ratios; separate broad-line group
    for AGN
\end{itemize}

\subsection{Forward Model}
\label{ssec:emline_model}

\begin{itemize}
  \item Intrinsic profile: log-Gaussian in observed wavelength
  \item DESI resolution matrix: pre-convolved representation with
    effective width
    $\sigma_\mathrm{eff} = \sqrt{\sigma^2 + \sigma_G^2}$
  \item Analytic Jacobian for efficient nonlinear least-squares
  \item Removes 5--30\% flux bias present in earlier bin-integration
    approach for unresolved lines
\end{itemize}

\subsection{Fitting Algorithm and Upper Limits}
\label{ssec:emline_fitting}

\begin{itemize}
  \item Nonlinear least-squares on continuum-subtracted residuals
  \item Significance thresholds and upper limit computation
  \item Equivalent widths from adjacent continuum flux densities measured
    from the SED model
\end{itemize}

\subsection{Broad-Line AGN Model}
\label{ssec:broadlines}

\begin{itemize}
  \item Additional broad kinematic component for permitted lines
  \item $\deltachisq$-based decision to retain or drop broad component
  \item Conditions for dropping: $\deltachisq > 0$; all broad amplitudes
    zero; $\sigma_\mathrm{broad} < \sigma_\mathrm{narrow}$
\end{itemize}

\section{Derived Physical Quantities}
\label{sec:quantities}
%% ============================================================

\subsection{Stellar Masses and Star-Formation Rates}
\label{ssec:mstar_sfr}

\begin{itemize}
  \item Stellar mass from NNLS coefficients and FSPS surviving mass
    fractions
  \item SFR from emission lines (\ha, \oii) and/or UV continuum
  \item $D_n(4000)$ from line-free best-fit model (Balogh et al.\ 1999
    definition)
\end{itemize}

\subsection{K-corrections and Rest-Frame Magnitudes}
\label{ssec:kcorr}

\begin{itemize}
  \item K-corrections by comparing synthesized observer- and rest-frame
    photometry from the best-fit SED
  \item Rest-frame absolute magnitudes from distance modulus +
    K-correction
  \item Monochromatic luminosities at 1450, 1500, 1700, 2800, 3000, and
    5100~\AA
\end{itemize}

\subsection{Spectral Indices and Equivalent Widths}
\label{ssec:indices}

\begin{itemize}
  \item EWs computed relative to local continuum flux densities from the
    SED model
  \item $D_n(4000)$ measured from both the best-fit model and the observed
    spectrum
\end{itemize}
